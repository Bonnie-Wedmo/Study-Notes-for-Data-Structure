# 判断一个数是不是回文数
## 法一：左右两边往里收缩，效率奇低
```c
bool isPalindrome(int x) {
    if (x < 0) return false;
    else
    {
        char str[32];
        sprintf(str,"%d",x);
        int n = strlen(str);
        int left = 0;
        int right = n-1;
        while(left <= right)
        {
            if (str[left] != str[right])
            {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
}
